App::log()

Automatically writes a string to a log file stamped with the current time located in __LOG_PATH__

static bool log (string $errmsg, [string $logfile = 'log.txt'])

example: 

App::log( 'user logged in' );
[edit]

App::config()

Returns a handle to the AppConfig object

static AppConfig config()

example: 

App::config()->uri;
[edit]

App::session()

Returns a handle to the Session object

static Session session()

example: 

App::session()->destroy();
[edit]

App::forward()

Sets the next URI once the current controller is finished executing

static void forward ([string $nextController = ], [ $args = array()], [int $method = __FORWARD_METHOD_URI__])

example: 

App::forward( 'login' );
[edit]

App::flash()

Adds a message to message stack

static void flash (string $msg, [ $type = __STATUS_MESSAGE__])

example: 

App::flash( 's:your record was saved!' );
[edit]

App::messages()

Returns array of messages by type

static array messages ([string $type = ])

example: 

foreach( App::messages() as $msg ) :
[edit]

App::uri()

Returns a URI based on the requested module and parameters

static string uri ([ $module = ], [ $args = array()], [string $destination = ], string $controller)

example: 

App::uri( 'home' );
[edit]

App::url()

Returns a URL based on the requested module and parameters

static string url ([string $module = ], [ $args = array()], [string $destination = ])

example: 

App::url( 'home' );
[edit]

App::urlWrapper()

Parses a URL and returns a friendly URL

static string urlWrapper (string $queryString)

example: 

App::urlWrapper( 'http://mysite.com/?module=index&id=4' );
